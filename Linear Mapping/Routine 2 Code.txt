import random
import numpy as np
import matplotlib.pyplot as plt

# Function to convert a matrix A into a unit lower triangular matrix
def unit_lower_triangular_matrix(A, n):
    L_comp = [[0 for j in range(n)] for i in range(n)]
    for i in range(n):
        for j in range(n):
            if j < i:
                L_comp[i][j] = int(A[i][j])
            elif j == i:
                L_comp[i][j] = 1
            else:
                L_comp[i][j] = 0
    return L_comp

# Function to map L to a single indexed 1D-array by compressing the row entries below the diagonal
def compressed_row_lower_triangular(L_comp, n):
    S = int((n * (n+1) / 2) - n)
    L_mapped = [[0] for i in range(S)]
    index = 0
    for i in range(n):
        for j in range(i):
            L_mapped[index] = L_comp[i][j]
            index += 1
    return L_mapped

# Function to perform matrix-vector multiplication
def compressed_matrix_vector_product(L_mapped, v, n):
    w_comp = [[0] for i in range(n)]
    index = 0
    for i in range(n):
        w_comp[i][0] += v[i][0]
        for j in range(i):
            w_comp[i][0] += L_mapped[index] * v[j][0]
            index += 1
    return w_comp


if __name__ == '__main__':
    print("-> Sample Test for Routine 2: ")
# Set dimension n for generating (nxn) matrix and (nx1) vector.
# Input lower and upper ranges to generate random number entries.
    n = int(input('Set Dimension n: '))

# Run routines for computed L and w
    L_comp = unit_lower_triangular_matrix(A, n)
    L_mapped = compressed_row_lower_triangular(L_comp, n)
    w_comp = compressed_matrix_vector_product(L_mapped, v, n)

# Use Python library numpy for true L and w
    L_true = np.tril(A)

# Display the results
