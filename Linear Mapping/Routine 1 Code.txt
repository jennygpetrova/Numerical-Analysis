import random
import numpy as np
import matplotlib.pyplot as plt

# Function to convert a matrix A into a lower triangular matrix
def lower_triangular_matrix(A, n):
    L_comp = [[0 for j in range(n)] for i in range(n)]
    for i in range(n):
        for j in range(n):
            if (j <= i):
                L_comp[i][j] = int(A[i][j])
            else:
                L_comp[i][j] = 0
    return L_comp

# Function to perform matrix-vector multiplication
def matrix_vector_product(L_comp, v, n):
    w_comp = [[0] for i in range(n)]
    for i in range(n):
        for j in range(n):
            w_comp[i][0] += L_comp[i][j] * v[j][0]
    return w_comp

# Function to compute error between computed and true result w
def compute_errors(w_comp, w_true):
    abs_error = int(np.linalg.norm(np.array(w_true) - np.array(w_comp)))
    rel_error = int(abs_error / np.linalg.norm(np.array(w_true)))
    return abs_error, rel_error
