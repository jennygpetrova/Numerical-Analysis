import random
import numpy as np
import matplotlib.pyplot as plt

# Function to convert a matrix A into a banded unit lower triangular matrix
def banded_unit_lower_triangular(A, n):
    L_comp = [[0 for j in range(n)] for i in range(n)]
    L_banded = [[0 for j in range(n-1)] for i in range(2)]
    for i in range(n):
        for j in range(n):
            if (i - j == 1):
                L_comp[i][j] = int(A[i][j])
                if i > 0:
                    L_banded[1][i-1] = L_comp[i][j]
            elif (i - j == 2):
                L_comp[i][j] = int(A[i][j])
                if i > 1:
                    L_banded[0][i-1] = L_comp[i][j]
            elif (j == i):
                L_comp[i][j] = 1
            else:
                L_comp[i][j] = 0
    return L_comp, L_banded

# Function to perform matrix-vector multiplication
import numpy as np


def banded_matrix_vector_product(L_banded, v, n):
    # Initialize the result vector with zeros
    w_comp = [0] * n

    for i in range(n):
        # The value from the main diagonal
        w_comp[i] += L_banded[0][i] * v[i][0]

        # The value from the lower diagonal (sub-diagonal)
        if i > 0:
            w_comp[i] += L_banded[0][i - 1] * v[i - 1][0]

        # The value from the second lower diagonal (if applicable)
        if i > 1:
            w_comp[i] += L_banded[1][i - 2] * v[i - 2][0]

    return w_comp
