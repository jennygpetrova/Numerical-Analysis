' M <-- LU '
''' 
The actions of this routine are controlled by the parameters set at the beginning. 
Tests are run for the specified subroutine for a range of values of the dimension n
(specified by user input). Python's library and commands will be used for comparison
between "true" answer and the computed answer from the subroutines.
'''

import random
import numpy as np
import matplotlib.pyplot as plt

# Function to convert a matrix A into a unit lower triangular matrix
def unit_lower_triangular_matrix(A, n):
    L_comp = [[0 for j in range(n)] for i in range(n)]
    for i in range(n):
        for j in range(n):
            if j < i:
                L_comp[i][j] = int(A[i][j])
            elif j == i:
                L_comp[i][j] = 1
            else:
                L_comp[i][j] = 0
    return L_comp

# Function to convert a matrix A into an upper triangular matrix
def upper_triangular_matrix(A, n):
    U_comp = [[0 for j in range(n)] for i in range(n)]
    for i in range(n):
        for j in range(n):
            if (j >= i):
                U_comp[i][j] = int(A[i][j])
            else:
                U_comp[i][j] = 0
    return U_comp

# Function to combine a upper triangular matrix and a unit lower triangular matrix into a matrix B
def lower_upper_combined_matrix(L_comp, U_comp, n):
    B = [[0 for j in range(n)] for i in range(n)]
    for i in range(n):
        for j in range(n):
            if j < i:
                B[i][j] = L_comp[i][j]
            else:
                B[i][j] = U_comp[i][j]
    return B

# Function to perform matrix-matrix multiplication on
def lower_upper_matrix_product(B, n):
    M = [[0 for j in range(n)] for i in range(n)]
    for i in range(n):
        for j in range(n):
            sum = 0
            for k in range(min(i, j) + 1):
                if i == k:
                    sum += 1 * B[k][j]
                else:
                    sum += B[i][k] * B[k][j]
            M[i][j] = sum
    return M

# Compute the difference between the computed and true matrices
def compute_errors(M_comp, M_true):
    M = M_comp - M_true
    M_fnorm = np.linalg.norm(M, ord='fro')  # Frobenius norm
    M_1norm = np.linalg.norm(M, ord=1)  # 1-norm (maximum absolute column sum)
    M_2norm = np.linalg.norm(M, ord=2)  # 2-norm (largest singular value)
    M_infnorm = np.linalg.norm(M, ord=np.inf)  # Infinity norm (maximum absolute row sum)

    return M_fnorm, M_1norm, M_2norm, M_infnorm
